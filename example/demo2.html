Demo6(配置onClick提示信息)

 $("#demo6").myPagination({
          cssStyle: 'jogger',
          ajax: {
            on: false,
            onClick: function(page) {
              ZENG.msgbox.show(" 正在加载" + page + "页，请稍后...", 6, 1000);
            }
          }
        });
